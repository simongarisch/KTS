fun add(vararg ns: Int): Int {
    var total: Int = 0
    for (n in ns) {
        total += n
    }
    return total

ExtensionsÂ¶

Kotlin allows extension functions and extension properties that allow adding new functionality to existing classes.

fun Int.double(): Int {
    return this * 2
}
2.double()
4
fun Int.isDivBy(vararg ns: Int): Boolean {
    for (n in ns) {
        if (this % n != 0) {
            return false
        }
    }
    return true
}
10.isDivBy(7, 11)
false
10.isDivBy(2, 5, 10)
true
